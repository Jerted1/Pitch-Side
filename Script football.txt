const API_KEY = 'd8111c3233a64fd7b09494daa6a3f0ee'; // ‚úÖ Replace this with your RapidAPI key
const NEWS_API_KEY = 'd8111c3233a64fd7b09494daa6a3f0ee'; // ‚úÖ Replace this with your NewsAPI key

const options = {
  method: 'GET',
  headers: {
    'X-RapidAPI-Key': API_KEY,
    'X-RapidAPI-Host': 'api-football-v1.p.rapidapi.com'
  }
};

// üì∞ Fetch Football News
fetch(`https://newsapi.org/v2/top-headlines?q=football&apiKey=${NEWS_API_KEY}`)
  .then(res => res.json())
  .then(data => {
    const newsDiv = document.getElementById('news-content');
    if (data.articles) {
      data.articles.slice(0, 5).forEach(article => {
        const div = document.createElement('div');
        div.className = 'card';
        div.innerHTML = `<strong>${article.title}</strong><br>${article.description}`;
        newsDiv.appendChild(div);
      });
    } else {
      newsDiv.innerHTML = '‚ö†Ô∏è No news found.';
    }
  })
  .catch(err => {
    console.error('News API error:', err);
    document.getElementById('news-content').innerText = '‚ö†Ô∏è Error loading news.';
  });

// üîÅ Fetch Transfers
fetch('https://api-football-v1.p.rapidapi.com/v3/transfers?team=33', options)
  .then(res => res.json())
  .then(data => {
    const transfersDiv = document.getElementById('transfers-content');
    data.response.slice(0, 5).forEach(transfer => {
      const div = document.createElement('div');
      div.className = 'card';
      div.innerHTML = `${transfer.player.name} ‚û°Ô∏è ${transfer.transfers[0].teams.out.name} to ${transfer.transfers[0].teams.in.name}`;
      transfersDiv.appendChild(div);
    });
  })
  .catch(err => {
    console.error('Transfer API error:', err);
    document.getElementById('transfers-content').innerText = '‚ö†Ô∏è Error loading transfers.';
  });

// üìÖ Fetch Upcoming Fixtures
fetch('https://api-football-v1.p.rapidapi.com/v3/fixtures?team=33&next=5', options)
  .then(res => res.json())
  .then(data => {
    const fixturesDiv = document.getElementById('fixtures-content');
    data.response.forEach(game => {
      const div = document.createElement('div');
      div.className = 'card';
      const { home, away } = game.teams;
      const date = new Date(game.fixture.date).toLocaleString();
      div.innerHTML = `${home.name} vs ${away.name} <br><small>${date}</small>`;
      fixturesDiv.appendChild(div);
    });
  })
  .catch(err => {
    console.error('Fixtures API error:', err);
    document.getElementById('fixtures-content').innerText = '‚ö†Ô∏è Error loading fixtures.';
  });
